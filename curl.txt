//GET
    //Display bonjour.txt
        curl -X GET http://localhost:8005/bonjour.txt               -> 200 OK

    chmod 000 bonjour.txt
    //No access rights
        curl -X GET http://localhost:8005/bonjour.txt               -> 403 Forbidden

    chmod 777 bonjour.txt
    //Page not found
        curl -X GET http://localhost:8005/page_not_found.txt        -> 404 Not Found

    index exists
    //index.html exists, so we display index.html
        curl -X GET http://localhost:8005/www/                      -> 200 : OK
    
    index not exists
    autoindex on
    //index.html doesn't not exist, and autoindex is on, so we display directory
        curl -X GET http://localhost:8005/www/                      -> 200 : OK

    index not exists
    autonindex off
    //index.html doesn't not exist, and autoindex is off, so we display 404 Not Found
        curl -X GET http://localhost:8005/www/                      -> 404 Not Found

    Redirection
    //www is a directory, so we create a redirection to /www/
        curl -X GET http://localhost:8005/www                       -> 301 Moved Permanently

    CGI
    //script.php is a cgi, so we execute a script (there is no datas here)
        curl -X GET http://localhost:8005/www/cgi/script.php                    -> 200 OK

    //script.php is a cgi, so we execute a script (datas are bonjour=hello)
        curl -X GET -d "bonjour=hello" http://localhost:8005/www/cgi/script.php     -> 200 OK



//POST
    CGI
        curl -X POST http://localhost:8005/www/cgi/script.php                                       -> 200 OK
        curl -X POST -d "data1=value1&data2=value2" http://localhost:8005/www/cgi/script.php        -> 200 OK
        curl -X POST -F "name=user" -F "password=test" http://localhost:8005/www/cgi/script.php     -> 200 OK

    FILE
    //Uploading the bonjour.txt file to /www/uploads/ (because in the location, upload_dir exists)
        curl -X POST -F "file=@bonjour.txt" http://localhost:8005/www/                            -> 200 OK
    //Same thing, but the file Mkaefile is a binary file
        curl -X POST -F "file=@Makefile" http://localhost:8005/www/                               -> 200 OK

    //Creating the hello.txt file
        curl -X POST -d "Hello tout le monde !" http://localhost:8005/www/hello.txt              -> 201 Created
    //Updating file hello.txt
        curl -X POST -d "Hello tout le monde !" http://localhost:8005/www/hello.txt              -> 200 OK
    //No file specified in the URL
        curl -X POST -d "bonjour tout le monde" http://localhost:8005/www/                  -> 403 Forbidden



//DELETE
    //We are deleting the bonjour.txt file
        curl -X DELETE http://localhost:8005/bonjour.txt    -> 200 OK
    // We are trying to delete a non-existent file
        curl -X DELETE http://localhost:8005/bonjour.txt    -> 404 Not Found
    // We are trying to delete a directory (to do this, you must delete all the files in the latter)
        curl -X DELETE http://localhost:8005/www/           -> 200 OK
